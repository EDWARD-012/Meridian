-- trg_orderitem_reduce_stock: AFTER INSERT on orders_orderitem
UPDATE catalog_product SET stock_quantity = stock_quantity - NEW.quantity
WHERE id = NEW.product_id AND stock_quantity >= NEW.quantity;

-- trg_order_status_track: AFTER UPDATE on orders_order
IF OLD.status != NEW.status THEN
  INSERT INTO orders_ordertracking (order_id, status, message, location, recorded_at)
  VALUES (NEW.id, NEW.status, CONCAT('Status updated to ', NEW.status), 'System', NOW(6));
END IF;
